1 define(function(sys) {
  2     /** 
  3      * sys is a singleton system class that has all the basic functionalities
  4      * @name sys
  5      * @class sys
  6      * @author Ishaan Puniani -- 11-10-2013
  7      * @description 
  8      * @modeified 
  9      * @requires 
 10      * @example
 11      * 
 12      *  sys.isEmpty(evnt);
 13      *  
 14      * @constructor singleton
 15      * @return Shared system object for basic functionalitiies
 16      */
 17     if (!sys) {
 18         sys = {};
 19     }
 20 
 21     /**
 22      * Method to verify the object is empty or null 
 23      * @name sys#isEmpty
 24      * @function isEmpty
 25      * @author Ishaan Puniani -- 11-10-2013
 26      * @description Method to verify the object is null or undefined or array has no element 
 27      * @example
 28      * 
 29      *  if(this.isEmpty(obj))
 30      *  {
 31      *      // do stuff here
 32      *  }
 33      *  
 34      *  @param {object} o object to validate
 35      */
 36     sys.isEmpty = function(o) {
 37         return (o === "") || (o.length === 0) || (this.isArray(o) && !o.length);
 38     };
 39     /**
 40      * Method to verify the object is empty or null 
 41      * @name sys#isNullOrEmpty
 42      * @function isNullOrEmpty
 43      * @author Ishaan Puniani -- 11-10-2013
 44      * @description Method to verify the object is null or undefined or array has no element 
 45      * @example
 46      * 
 47      *  if(this.isNullOrEmpty(obj))
 48      *  {
 49      *      // do stuff here
 50      *  }
 51      *  
 52      *  @param {object} o object to validate
 53      */
 54     sys.isNullOrEmpty = function(o) {
 55         var nullOrEmpt =  true;
 56         if(this.isDefined(o)){
 57             nullOrEmpt= this.isEmpty(o);
 58         }
 59         return nullOrEmpt;
 60     };
 61     /**
 62      * Method to verify the object is not undefined
 63      * @name sys#isDefined
 64      * @function isDefined
 65      * @author Ishaan Puniani -- 11-10-2013
 66      * @description Method to verify the object is not undefined
 67      * @example
 68      * 
 69      *  if(this.isDefined(obj))
 70      *  {
 71      *      // do stuff here
 72      *  }
 73      *  
 74      *  @param {object} o object to validate
 75      */
 76     sys.isDefined = function(o) {
 77         // checking typeof is more secure than just comparing type, since this also handles
 78         // situations where o is not declared
 79         return (typeof o !== "undefined" &&  o !== null);
 80     };
 81     /**
 82      * Method to verify the object is an Array
 83      * @name sys#isArray
 84      * @function isArray
 85      * @author Ishaan Puniani -- 11-10-2013
 86      * @description Method to verify the object is an Array
 87      * @example
 88      * 
 89      *  if(this.isArray(obj))
 90      *  {
 91      *      // do stuff here
 92      *  }
 93      *  
 94      *  @param {object} arr object to validate
 95      */
 96     sys.isArray = function(arr) {
 97         var type = window.toString.apply(arr);
 98         return (type === '[object Array]' || type === '[object NodeList]' || type === '[object TouchList]');
 99     };
100     /**
101      * Method to verify the object is a string
102      * @name sys#isString
103      * @function isString
104      * @author Ishaan Puniani -- 11-10-2013
105      * @description Method to verify the object is String
106      * @example
107      * 
108      *  if(this.isString(obj))
109      *  {
110      *      // do stuff here
111      *  }
112      *  
113      *  @param {object} str object to validate
114      */
115     sys.isString = function(str) {
116         return !this.isArray(str) && (typeof str === "string");
117     };
118     /**
119      * Method to verify the object is an Object
120      * @name sys#isObj
121      * @function isObj
122      * @author Ishaan Puniani -- 11-10-2013
123      * @description Method to verify the object is an Object and not an Array
124      * @example
125      * 
126      *  if(this.isObj(obj))
127      *  {
128      *      // do stuff here
129      *  }
130      *  
131      *  @param {object} obj object to validate
132      */
133     sys.isObj = function(obj) {
134         return !this.isArray(obj) && (typeof obj === "object");
135     };
136     return sys;
137 });
