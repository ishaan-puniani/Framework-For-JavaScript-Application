1 define(["base/observable"], function(observable){
  2    /** 
  3      * User Module that represent the user in tha application.
  4      * @name user
  5      * @class user
  6      * @author Ishaan Puniani -- 11-10-2013
  7      * @description A user is an module that subscribe to say hello event.
  8      *  It adds a subscriber i.e. it tells the framework that i am interested in linstening to this event
  9      *  and framework will invoke this method when this event is fired
 10      * @extends observable
 11      * @modeified 
 12      * @requires "base/observable"
 13      * @example
 14      * 
 15      * var player1 = new user();
 16      *
 17      * @constructor init
 18      * @return user
 19      */
 20     var user = observable.extend({
 21         init:function(){
 22            this._super( );
 23            
 24            // add a subscriber i.e. it tells the framework that i am interested in linstening to this event
 25            // and framework will invoke this method when this event is fired
 26            this.subscribeEvent("btnSayHello:Click",function(){
 27                alert('btnSayHello:Click is subscribed from user module');
 28            }); 
 29         }
 30     });
 31     
 32     return user;
 33 });
