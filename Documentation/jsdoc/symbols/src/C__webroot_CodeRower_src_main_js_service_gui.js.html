1 define(["base/observable","base/sys"], function(observable,sys){
  2    /** 
  3      * gui module is ment for DOM elements and their manipulation.
  4      * @name gui
  5      * @class gui
  6      * @author Ishaan Puniani -- 11-10-2013
  7      * @description GUI is very important part of the framework and this module is prefered over native DOM manipulation is fue to the fact that now a days most of the applications are build to support multiple devices.
  8      *              So, its important to handle the gestures provided by the specific device as efficiently as possible
  9      * @extends observable
 10      * @modeified 
 11      * @requires "base/observable"
 12      * @example
 13      * var child = observable.extend({
 14      *
 15      *    });
 16      * @constructor init
 17      * @return gui element instance
 18      */
 19 var gui = observable.extend({
 20     init:function(obj){
 21         var me = this,el;
 22         // shorthand to this.isDefined
 23             var isDef = sys.isDefined;
 24         
 25         // if obj is a string, use that for tag name in the DOM element
 26         if(sys.isString(obj)) {
 27             el = document.createElement(obj);
 28         }
 29         
 30         // else, if obj is an HTML DOM element, connect to that
 31         // else, if obj is an object, use it to make more detailed DOM element
 32         else if(obj !== null && sys.isObj(obj)) {
 33             if (/HTML[a-zA-z]*Element/i.test(obj.toString())) {
 34                 el = obj;
 35             }
 36             else {
 37 
 38                 el = document.createElement(obj.tag);
 39 
 40                 // if class is defined
 41                 if(isDef(obj.id)) {
 42                     el.id = obj.id;
 43                 }
 44 
 45                 // if class is defined
 46                 if(isDef(obj.cls)) {
 47                     el.className = obj.cls;
 48                 }
 49 
 50                 // if inner html is defined
 51                 if(isDef(obj.innerHTML)) {
 52                     el.innerHTML = obj.innerHTML;
 53                 }
 54 
 55                 if(isDef(obj.innerText)) {
 56                     el.innerText = obj.innerText;
 57                 }
 58 
 59                 // if src is defined
 60                 if(isDef(obj.src)) {
 61                     el.src = obj.src;
 62                 }
 63 
 64                 if(isDef(obj.style)) {
 65                     el.style.cssText = obj.style;
 66                 }
 67 
 68                 // if clicklable object is defined
 69                 if (isDef(obj.clickable)) {
 70                     me.onClickEvent = obj.clickable.onClickEvent;
 71                         el.addEventListener("click",function(e){
 72                            me.onClickEvent();
 73                         },false);
 74                 }
 75             }
 76             // if not string or object, throw exception
 77         }
 78         else{
 79             throw "Invalid instantiation of Sys.Element, invalid input, needs element string or object";
 80         }
 81         if(!sys.isEmpty(obj.renderTo)) {
 82             if(obj.renderTo === "body"){
 83              document.getElementsByTagName("body")[0].appendChild(el);   
 84             }else{
 85             document.getElementById(obj.renderTo).appendChild(el);
 86             }
 87         }
 88 
 89         me.el = el;
 90 
 91         
 92         this._super( );
 93     }
 94 });
 95 return gui;
 96 });
