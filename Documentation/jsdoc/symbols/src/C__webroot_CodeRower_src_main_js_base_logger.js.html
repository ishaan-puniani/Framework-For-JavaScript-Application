1 define(function() {
  2 
  3     /** 
  4      * Application Logger 
  5      * @name logger
  6      * @class logger
  7      * @author Ishaan Puniani -- 11-10-2013
  8      * @description Logger top log all kind of errors , info messages etc.
  9      * @extends 
 10      * @modeified 
 11      * @requires 
 12      * @example
 13      * 
 14      * logger.log("this is logged");
 15      * 
 16      * @constructor init
 17      * @return 
 18      */
 19     var logger = {
 20         /**
 21          * Constant field to add type information while logging 
 22          * @name logger#messageType
 23          * @constant Message Type that is going to be logged
 24          * @author Ishaan Puniani -- 11-10-2013
 25          * @description Type of Message to be logged like "Error-", "Information-"
 26          * @example
 27          * 
 28          *  logger.messageType.info
 29          *  logger.messageType.error
 30          *  logger.messageType.evntSubscribe
 31          *  logger.messageType.evntPublish
 32          *  
 33          */
 34         messageType: {
 35             info: "Information",
 36             error: "Error",
 37             evntSubscribe: "SubscribeEvent",
 38             evntPublish: "PublishEvent"
 39         },
 40         /**
 41          * Method to log messages
 42          * @name logger#log
 43          * @function log
 44          * @author Ishaan Puniani -- 11-10-2013
 45          * @description Method to log messages, Errors etc
 46          * @example
 47          * 
 48          * logger.log(logger.messageType.evntPublish,evnt);
 49          *  
 50          *  @param {string} type message type to be logged
 51          *  @param {string} txtMessage Message to be logged
 52          */
 53         log: function(type, txtMessage) {
 54             console.log(type + " - " + txtMessage);
 55         },
 56         /**
 57          * Method to log errors
 58          * @name logger#logError
 59          * @function logError
 60          * @author Ishaan Puniani -- 11-10-2013
 61          * @description Method to log Errors 
 62          * @example
 63          * 
 64          * logger.logError(error message);
 65          *  
 66          *  @param {string} message Message to be logged
 67          */
 68         logError: function(message) {
 69             this.log(this.messageType.error, message);
 70         },
 71         /**
 72          * Method to log Information
 73          * @name logger#logInfo
 74          * @function logInfo
 75          * @author Ishaan Puniani -- 11-10-2013
 76          * @description Method to log Information
 77          * @example
 78          * 
 79          * logger.logInfo(info);
 80          *  
 81          *  @param {string} info Information to be logged
 82          */
 83         logInfo: function(info) {
 84             this.log(this.messageType.info, info);
 85         }
 86     };
 87 
 88     return logger;
 89 });
